PARENT_A_PROGRAM_EXAMPLE:
```python
import random, sympy

def generate(seed):
    rng = random.Random(seed)
    while True:
        S = rng.randint(10, 80)
        if S % 2 != 0:
            continue
        if S // 2 < 5:
            continue
        break
    problem = (
        f"Two positive real numbers x and y satisfy x + y = {S}. "
        f"Compute their arithmetic mean (x + y)/2."
    )
    return problem, str(sympy.Integer(S // 2))
```

PARENT_B_PROGRAM_EXAMPLE:
```python
import random, sympy

def generate(seed):
    rng = random.Random(seed)
    while True:
        pairs = [(rng.randint(1, 9), rng.randint(1, 9)) for _ in range(4)]
        products = [a * b for a, b in pairs]
        if len(set(products)) < len(products):
            continue
        best = max(products)
        if best < 10:
            continue
        break
    listing = ", ".join(f"({a}, {b})" for a, b in pairs)
    problem = (
        f"Among the following pairs of positive integers, find the "
        f"largest value of the product x*y: {listing}."
    )
    return problem, str(sympy.Integer(best))

CONCEPT_GROUP = "algebra"
CONCEPT_TYPE = "algebra.product_extremum_discrete"
```

CROSSOVER_CHILD_PROGRAM_EXAMPLE:
```python
"""Crossover hybrid generator.

Parents fused: arithmetic_mean (a fixed sum x + y = S) and
product_extremum_discrete (maximising x*y). They are not chained;
the AM-GM inequality (x + y)/2 >= sqrt(x*y) makes the fixed sum
bound the product directly.

To solve, given only S, the solver must:
  (1) see that x*y must be maximised over a continuous set, so
      no "scan a list" shortcut applies,
  (2) recall AM-GM and read off that x*y <= (S/2)^2,
  (3) note the bound is tight at x = y and report (S/2)^2.
Without AM-GM there is no shorter handle on a continuous optimum.

Guards keep the task honest: S even (integer answer), (S/2)^2 > 25
(not trivially small), S >= 8 (optimum away from a guessable
boundary). The problem text states only S — never AM-GM, the
x = y optimum, or completing the square.
"""

import random, sympy

def generate(seed):
    rng = random.Random(seed)
    while True:
        S = rng.randint(8, 60)
        if S % 2 != 0:
            continue
        best = (S // 2) ** 2
        if best <= 25:
            continue

        scan = max(x * (S - x) for x in range(1, S))
        assert scan == best
        break

    problem = (
        f"Let x and y be positive real numbers with x + y = {S}. "
        f"Find the maximum possible value of the product x*y."
    )
    return problem, str(sympy.Integer(best))

CONCEPT_REASON = "To solve, the solver must recognise a product under a fixed-sum constraint is maximised via the AM-GM inequality, tight at x = y — that inequality is the problem's reasoning core."
CONCEPT_GROUP = "algebra"
CONCEPT_TYPE = "algebra.am_gm_product_max"
```
