PARENT_A_PROGRAM_EXAMPLE:
```python
import math
import sympy
import random

def generate(seed):
    rng = random.Random(seed)
    while True:
        n = rng.randint(8, 16)
        k = rng.randint(3, n - 3)
        if k == n - k:
            continue
        answer = math.comb(n, k)
        if answer < 30:
            continue
        break
    problem = (
        f"Compute the binomial coefficient C({n}, {k}), i.e. the "
        f"number of ways to choose {k} items from {n} distinct items."
    )
    return problem, str(sympy.Integer(answer))

CONCEPT_GROUP = "combinatorics"
CONCEPT_TYPE = "combinatorics.binomial_coefficient"
```

PARENT_B_PROGRAM_EXAMPLE:
```python
import sympy
import random

def generate(seed):
    rng = random.Random(seed)
    while True:
        r = rng.choice([2, 3, 4, 5])
        n = rng.randint(4, 8)
        if r == 2 and n < 5:
            continue
        total = (r ** n - 1) // (r - 1)
        if total < 50:
            continue
        break
    problem = (
        f"Compute the sum of the geometric series "
        f"1 + {r} + {r}^2 + ... + {r}^{n - 1}."
    )
    return problem, str(sympy.Integer(total))

CONCEPT_GROUP = "sequence"
CONCEPT_TYPE = "sequence.geometric_sum"
```

CROSSOVER_CHILD_PROGRAM_EXAMPLE:
```python
"""Crossover hybrid generator.

Parents fused: binomial_coefficient (the C(n, k) counts) and
geometric_sum (a progression of powers of r). The binomial
theorem (1 + r)^n = sum_{k=0..n} C(n, k) * r^k puts both on one
polynomial — the binomials are exactly the weights of the sum.

Surface form is INVERTED relative to either parent: instead of
listing the weighted series and asking for its value, we give
the collapsed value V = (1 + r)^n together with the base r, and
ask for the exponent n. This is one fusion among many — the same
two concepts could instead be posed as a coefficient-extraction,
a mode-of-weights question, etc. Here we deliberately avoid the
"sum these terms" template so the child does not inherit either
parent's phrasing.

To solve, the solver must:
  (1) read V as a value that factors as a single power of (1 + r),
  (2) recognise (via the binomial theorem) that V is exactly the
      collapsed weighted sum sum_k C(n,k) r^k,
  (3) recover n = log_{1+r}(V).
The binomial theorem is what links the geometric base r to the
binomial weights; without it the connection between V and the
two parent concepts is invisible.

Guards keep the task honest: r != 1 (base 1+r must be >= 3 so the
log is unambiguous), n in a range where V is large enough not to
be guessable, and we never reveal n, the exponent relation, or
the binomial theorem in the problem text.
"""

import sympy
import random

def generate(seed):
    rng = random.Random(seed)
    while True:
        r = rng.choice([2, 3, 4, 5])
        n = rng.randint(6, 11)

        base = 1 + r
        value = base ** n
        if value < 10_000:          # large enough to discourage guessing
            continue
        # answer must be the unique exponent: base >= 3 guarantees this
        if base < 3:
            continue

        # sanity: V really is the collapsed binomial-weighted sum
        check = sum(sympy.binomial(n, k) * r ** k for k in range(n + 1))
        assert int(check) == value
        break

    problem = (
        f"A positive integer V equals the value of the sum "
        f"S = C(n,0) + C(n,1)*{r}^1 + C(n,2)*{r}^2 + ... + C(n,n)*{r}^n, "
        f"where C(n, k) denotes the binomial coefficient and n is a "
        f"positive integer. Given that V = {value}, determine n. "
        f"State the answer as a single integer."
    )
    return problem, str(sympy.Integer(n))

CONCEPT_REASON = "The solver must recognise the binomially-weighted geometric sum as (1 + r)^n via the binomial theorem, then invert that power to recover the exponent n — the recognition-and-inversion is the reasoning core."
CONCEPT_GROUP = "combinatorics"
CONCEPT_TYPE = "combinatorics.binomial_theorem_inverse"
```
