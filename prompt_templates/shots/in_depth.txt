CHILD_PROGRAM_EXAMPLE_1:
```python
"""In-depth mutation generator (uncertainty-raised + cross-checked).

Deepens modular_reduction: the modulus m may be prime or
composite, so the solver must first decide between Fermat and
Euler before reducing the exponent and finishing with the
parent's mod step.

To solve, given only a, b, m, the solver must:
  (1) see that forming a^b directly is hopeless,
  (2) DECIDE the tool: m prime -> Fermat (exp mod m-1);
      m composite with gcd(a,m)=1 -> Euler (exp mod phi(m)),
  (3) reduce the exponent with the correct totient,
  (4) reduce a^(b') mod m — the parent's original move.
Applying Fermat blindly to a composite m gives a wrong answer.

Two independent answer paths are cross-checked at generation
time; a mismatch rejects the candidate, so no mislabeled problem
can ship:
  PATH A (insight): reduce exponent via totient, then pow.
  PATH B (brute):   pow(a, b, m) over the full exponent.

Guards: m prime ~half the time, composite (coprime to a)
otherwise; b' not in {0,1}; residue not in {0,1,m-1}; gcd(a,m)=1.
Problem text states only a, b, m — never the theorem or that m
may be composite.
"""

import random
import sympy

def generate(seed):
    rng = random.Random(seed)
    while True:
        if rng.random() < 0.5:
            m = rng.choice([7, 11, 13, 17, 19, 23])
        else:
            m = rng.choice([12, 15, 21, 22, 25, 33, 35])

        a = rng.randint(2, m - 1)
        if sympy.gcd(a, m) != 1:
            continue

        b = rng.randint(m ** 2, m ** 3)

        # --- PATH A: insight (totient exponent reduction) ---
        phi = sympy.totient(m)
        b_reduced = int(b % phi)
        if b_reduced in (0, 1):
            continue
        answer_insight = pow(a, b_reduced, m)

        # --- PATH B: brute (full exponent, independent) ---
        answer_brute = pow(a, b, m)

        # --- CROSS-CHECK: insight path must equal brute path ---
        # catches a wrong totient / mis-selected theorem at gen time
        assert answer_insight == answer_brute, (
            f"path mismatch m={m} a={a} b={b}: "
            f"insight={answer_insight} brute={answer_brute}"
        )

        # independent confirmation of the prime/composite branch:
        # Fermat is valid iff m is prime, in which case phi == m-1
        assert (sympy.isprime(m) == (phi == m - 1))

        answer = answer_insight
        if answer in (0, 1, m - 1):
            continue
        break

    problem = (
        f"Compute the remainder when {a}^{b} is divided by {m}. "
        f"State only the remainder as an integer between 0 and {m - 1}."
    )
    return problem, str(sympy.Integer(answer))

CONCEPT_REASON = "To solve, the solver must first decide whether the modulus is prime (Fermat) or composite (Euler), compute the correct totient to reduce the exponent, then perform the final mod reduction — that tool selection plus exponent reduction is the reasoning core."
CONCEPT_GROUP = "number_theory"
CONCEPT_TYPE = "number_theory.modular_reduction"
```

```python
import random
import sympy

def generate(seed):
    rng = random.Random(seed)
    while True:
        x = sympy.symbols('x')
        deg = rng.randint(2, 3)
        coeffs = [rng.randint(-5, 5) for _ in range(deg + 1)]
        if coeffs[0] == 0:
            continue
        c = rng.randint(-4, 4)
        if c == 0:
            continue
        poly = sum(coeffs[i] * x ** (deg - i) for i in range(deg + 1))
        answer = int(poly.subs(x, c))
        if abs(answer) < 5:
            continue
        break
    terms = []
    for i, k in enumerate(coeffs):
        power = deg - i
        if k == 0:
            continue
        if power == 0:
            terms.append(f"{k:+d}")
        elif power == 1:
            terms.append(f"{k:+d}*x")
        else:
            terms.append(f"{k:+d}*x^{power}")
    expr = " ".join(terms).lstrip("+ ").replace("+ -", "- ")
    problem = (
        f"Let p(x) = {expr}. Compute p({c})."
    )
    return problem, str(sympy.Integer(answer))

CONCEPT_GROUP = "algebra"
CONCEPT_TYPE = "algebra.polynomial_evaluation"
```

CHILD_PROGRAM_EXAMPLE_2:
```python
"""In-depth mutation generator (polynomial evaluation via finite differences + cross-check).

Deepens polynomial_evaluation: instead of giving p(x) explicitly, the
problem gives the first few values of a hidden polynomial sequence and asks
for a later value. The solver must recognize that the sequence comes from
a low-degree polynomial and use finite differences to extrapolate.

To solve, given p(0), p(1), ..., p(d), and the target n, the solver must:
  (1) infer that the listed values determine a polynomial sequence of degree d;
  (2) compute the finite-difference table;
  (3) use Newton's forward formula to evaluate p(n);
  (4) return the single integer value.

This is a deeper variant of polynomial evaluation because the polynomial is
not displayed in coefficient form. The solver must reconstruct the evaluation
structure from values rather than directly substitute x into an explicit
formula.

Two independent answer paths are cross-checked at generation time:
  PATH A (insight): finite differences + Newton forward formula.
  PATH B (brute): construct the hidden polynomial explicitly and substitute n.

Guards ensure the sequence is non-degenerate, the degree is genuine, the
target is far enough from the given data to prevent inspection, and the
answer is not trivially small.
"""

import random
import sympy

def generate(seed):
    rng = random.Random(seed)
    x = sympy.symbols("x")

    while True:
        deg = rng.randint(3, 5)

        coeffs = [rng.randint(-4, 4) for _ in range(deg + 1)]
        if coeffs[0] == 0:
            continue

        poly = sum(coeffs[i] * x ** (deg - i) for i in range(deg + 1))

        given_values = [int(poly.subs(x, k)) for k in range(deg + 1)]

        # Guard against visually trivial or nearly constant sequences.
        if len(set(given_values)) <= deg:
            continue
        if max(abs(v) for v in given_values) < 10:
            continue

        n = rng.randint(deg + 4, deg + 12)

        # --- PATH A: insight (finite differences + Newton forward formula) ---
        layers = [given_values[:]]
        while len(layers[-1]) > 1:
            prev = layers[-1]
            layers.append([prev[i + 1] - prev[i] for i in range(len(prev) - 1)])

        first_differences = [layer[0] for layer in layers]
        answer_insight = 0
        for k, delta_k in enumerate(first_differences):
            answer_insight += sympy.binomial(n, k) * delta_k
        answer_insight = int(answer_insight)

        # --- PATH B: brute (explicit hidden polynomial substitution) ---
        answer_brute = int(poly.subs(x, n))

        # --- CROSS-CHECK: finite-difference route must equal direct polynomial route ---
        assert int(answer_insight) == int(answer_brute), (
            f"path mismatch deg={deg} n={n}: "
            f"insight={answer_insight} brute={answer_brute}"
        )

        # Confirm the generated sequence genuinely has the intended degree.
        assert layers[-1][0] != 0

        answer = answer_insight
        if abs(answer) < 20:
            continue
        if answer in given_values:
            continue
        break

    value_text = ", ".join(str(v) for v in given_values)
    problem = (
        f"A sequence a_0, a_1, ..., a_{deg} begins with the values "
        f"{value_text}. These values come from evaluating a polynomial p(t) "
        f"of degree at most {deg} at t = 0, 1, ..., {deg}, respectively. "
        f"Compute p({n}). State only the integer value."
    )
    return problem, str(sympy.Integer(answer))

CONCEPT_REASON = "To solve, the solver must recognize that the listed consecutive values determine a polynomial sequence, build the finite-difference table, and use Newton forward extrapolation to evaluate the polynomial at a later integer input."
CONCEPT_GROUP = "algebra"
CONCEPT_TYPE = "algebra.polynomial_evaluation"
```
